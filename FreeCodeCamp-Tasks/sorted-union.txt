function uniteUnique(...arr) {
  let newArr =  [];
  let result = [];
  for(let i=0 ; i < arr.length ; i++){
      for(let j=0 ; j < arr[i].length ; j++){
        newArr.push(arr[i][j]);
      }
  }
  for(let x=0 ; x < newArr.length ; x++){
      if(newArr.indexOf(newArr[x]) == x){
          result.push(newArr[x]);
      }
  }
  return result;
}

uniteUnique([1, 3, 2], [5, 2, 1, 4], [2, 1]);
